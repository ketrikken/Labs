int ChekIntAns(std::string ans, bool& eror)
{
	int IntAns = 0;
	bool sign = 0;
	int dec = 1;
	for (int i = ans.length()-1; i >=0; --i)
	{
		
		if (ans[i] == '-' && i == 0) sign = 1;
		else
		{
			if ((int)ans[i] >= 48 && (int)ans[i] <= 57)
			{
				IntAns += ((int)ans[i] - 48) * dec;
				dec *= 10;
			}
			else
			{
				eror = 1;
				return 0;
			}
		}
	}
	if (sign) IntAns *= -1;
	return IntAns;
}
